```python
"""
innate.py — v6.0 先天反射/偏好定义

零预训练模式下 Agent 的"出厂配置"。
不包含任何语言知识——只有身体稳态反射、注意力偏好和可塑性参数。
这些是人类婴儿出生时就具备的先天结构。

对应脑区:
  - hypothalamus: 身体稳态反射 (饥饿/渴/痛)
  - locus_coeruleus: 新颖性偏好 (NE 基线)
  - FPN: 注意力的先天偏向 (突然的声音/运动)
  - hippocampus: Hebb 可塑性参数 (学习率/遗忘率)
"""

import numpy as np


# ============================================================
# 纯净模式 Theta 调整（适配零预训练）
# ============================================================

PURE_MODE_THETA_OVERRIDES = {
    'cluster_threshold': 0.55,   # 降低匹配阈值（初始簇少, 需放宽 → 更快形成概念）
    'learn_rate_l0': 0.12,       # 提高初始学习率（快速吸收新知识，类似婴儿关键期）
    'decay_rate': 0.005,         # 降低衰减（避免早期记忆过快消失）
    'exploration_bonus': 0.3,    # 提高探索（鼓励主动学习，新颖性寻求）
    'critical_window': 5000,     # 延长关键期（语言学习敏感期）
    'w_social': 2.0,             # 增强社会驱力（婴儿对照顾者的社会依附）
    'temperature': 1.5,          # 提高温度（更随机的探索行为）
}

# ============================================================
# 身体稳态先天设定点 (文本对话模式, M=9)
# ============================================================

INNATE_SETPOINTS = np.array([
    0.65,   # b[0] 社交需求 — 对人类互动的先天渴望
    0.70,   # b[1] 能量/安全 — 需要充足能量
    0.05,   # b[2] 压力/疲劳 — 初始无压力
    0.20,   # b[3] 新颖性寻求 — 较高的先天好奇
    0.40,   # b[4] 专注/警觉 — 中等警觉
    0.30,   # b[5] 视觉刺激 — 初始低
    0.30,   # b[6] 听觉刺激 — 初始低
    0.50,   # b[7] 认知负荷 — 中等
    0.90,   # b[8] 组织完整性 — 健康
], dtype=np.float32)

INNATE_DECAYS = np.array([
    -0.004,  # b[0] 社交需求缓慢下降（孤独累积）
     0.000,  # b[1] 能量自然平衡
     0.002,  # b[2] 压力缓慢累积
     0.001,  # b[3] 新颖性缓慢漂移
     0.000,  # b[4] 专注自然平衡
    -0.003,  # b[5] 视觉刺激衰减
    -0.003,  # b[6] 听觉刺激衰减
     0.001,  # b[7] 认知负荷缓慢累积
     0.001,  # b[8] 组织完整性缓慢自愈
], dtype=np.float32)

# ============================================================
# 注意力先天偏向
# ============================================================

# 哪些感知维度天生更受注意（婴儿：突然声音 > 静止图像 > 文字）
INNATE_ATTENTION_BIAS = np.zeros(516, dtype=np.float32)
INNATE_ATTENTION_BIAS[0:64] = 1.0     # 文字（对话中是主要输入）
INNATE_ATTENTION_BIAS[372:468] = 0.6  # 听觉通道（突然声音吸引注意）
INNATE_ATTENTION_BIAS[64:372] = 0.4   # 视觉通道（运动吸引注意）


def apply_innate_config(agent) -> None:
    """将先天配置应用到 Agent 实例。

    在 Agent.__init__() 之后调用，覆盖默认参数为纯净模式参数。
    不影响 None/default 检查 — 只设置应该被覆盖的值。

    Args:
        agent: Agent 实例 (已调用 __init__)
    """
    # Theta 覆盖
    for key, value in PURE_MODE_THETA_OVERRIDES.items():
        if hasattr(agent.theta, key):
            setattr(agent.theta, key, value)

    # 身体设定点
    if agent.body is not None:
        M = len(agent.body.b)
        if M <= len(INNATE_SETPOINTS):
            agent.body.setpoints = INNATE_SETPOINTS[:M].copy()
            agent.body.decays = INNATE_DECAYS[:M].copy()
        else:
            # 扩展模式（未来更多维度）
            agent.body.setpoints = INNATE_SETPOINTS.copy()
            agent.body.decays = INNATE_DECAYS.copy()

    # FPN 注意力先天偏向
    if hasattr(agent, 'fpn') and agent.fpn is not None:
        # 将先天偏向混入初始注意力模板
        agent.fpn.attention_template = (
            0.7 * agent.fpn.attention_template
            + 0.3 * INNATE_ATTENTION_BIAS
        )


def get_innate_reflexes() -> dict:
    """返回先天反射的元数据（供调试/诊断使用）。

    这些不是可调参数，而是 Agent "出厂"时的硬编码属性。
    """
    return {
        'pure_mode': True,
        'theta_overrides': PURE_MODE_THETA_OVERRIDES,
        'M_body': 9,
        'attention_bias_summary': {
            'text': float(INNATE_ATTENTION_BIAS[0:64].mean()),
            'vision': float(INNATE_ATTENTION_BIAS[64:372].mean()),
            'audio': float(INNATE_ATTENTION_BIAS[372:468].mean()),
        },
    }
```

```python
"""
semantic_memory.py — 语义记忆系统 (前颞叶/角回等价物)

脑区映射:
  - 前颞叶 (Anterior Temporal Lobe): 跨模态概念整合, 语义中枢
  - 角回 (Angular Gyrus BA39): 语义知识的分布式存储
  - 部分顶下叶 + 额下回: 语义提取通路

功能:
  - 存储"知道是什么"的知识 (与情景记忆的"记得经历过"区分)
  - 从多次情景记忆中提取主旨性知识
  - 慢学习率 → 需要多次重复才能固化
  - 慢衰减率 → 一旦形成不易遗忘
  - 大容量 → 可存储上千个概念/事实

与海马情景记忆的关键差异:
  ┌──────────────┬─────────────────┬─────────────────┐
  │              │ 情景记忆 (海马)  │ 语义记忆 (皮层)  │
  ├──────────────┼─────────────────┼─────────────────┤
  │ 学习速度     │ 快 (lr=0.05)    │ 慢 (lr=0.01)    │
  │ 遗忘速度     │ 快 (decay=0.02) │ 慢 (decay=0.003)│
  │ 容量         │ 512 簇          │ 1024 簇         │
  │ 存储内容     │ 完整512维体验   │ 64维主旨向量    │
  │ 匹配阈值     │ 中等 (0.70)     │ 低 (0.45)       │
  │ 主观体验     │ "记得"          │ "知道"           │
  └──────────────┴─────────────────┴─────────────────┘
"""

import numpy as np
from typing import Optional, List, Tuple
from cns.data_types import D, Theta, Cluster
from cerebrum.limbic_system.hippocampus import (
    ClusterNetwork, _masked_cosine, _auto_mask,
)


# 语义记忆专属参数
SEMANTIC_THETA_DEFAULTS = {
    'cluster_threshold': 0.45,   # 低阈值 — 相似概念合并 (比情景宽)
    'learn_rate_l0': 0.01,       # 慢学习 — 需要多次重复 (比情景慢5x)
    'decay_rate': 0.003,         # 慢遗忘 — 知识一旦形成不易消失
}


class SemanticMemory:
    """皮层语义记忆 — "知道"型知识的独立 ClusterNetwork。

    与海马情景记忆使用相同的 ClusterNetwork 配方，
    但参数调校不同：慢学、慢衰、低阈值、大容量。

    存储格式 (D=516 维质心):
      centroid[:64]   = text主旨   — 概念/事实的语义编码
      centroid[64:72] = body_snap  — 编码时的身体状态快照 (用于情境提取)
      centroid[72:80] = emotion    — 效价/唤醒/情感标签
      centroid[80:]   = 保留       — 跨模态关联 (视觉/听觉/痛觉简化为0)
    """

    def __init__(self, max_clusters: int = 1024):
        theta = Theta(**SEMANTIC_THETA_DEFAULTS)
        self.net = ClusterNetwork(theta, hash_offset=0)
        self.max_clusters = max_clusters
        self.n_facts: int = 0           # 累计存储的事实/概念数
        self.consolidation_count: int = 0  # 系统巩固次数

    # ----- 提取主旨向量 -----
    def extract_gist(self, episodic_centroid: np.ndarray) -> np.ndarray:
        """从完整情景记忆 (516维) 中提取主旨 (64维语义)。

        只保留: text[0:64] + body_snap[64:72] + emotion[72:80]
        丢弃: 视觉细节[64:372]、听觉细节[372:468]、痛觉细节[468:516]

        这模拟了记忆巩固中"具体细节褪去, 主旨保留"的过程。
        """
        gist = np.zeros(D, dtype=np.float32)
        # 文本主旨
        gist[:64] = episodic_centroid[:64]
        # 身体快照 (压缩为 8 维)
        if len(episodic_centroid) > 64:
            body_slice = episodic_centroid[64:72]
            gist[64:72] = body_slice[:8]
        # 情感快照 (从感知向量中提取)
        if len(episodic_centroid) > 80:
            gist[72:80] = episodic_centroid[72:80]
        # 其余维度保留为零 (不编码通道细节)
        return gist

    # ----- 学习/存储 -----
    def learn_fact(self, fact_vec: np.ndarray, weight: float = 1.0) -> Cluster:
        """学习一个事实/概念。

        使用低学习率 — 单次暴露不足以形成稳定语义知识。
        需要多次重复或高权重才能固化。

        Args:
            fact_vec: (D,) 全维度向量 (已通过 extract_gist 或直接输入)
            weight: 学习权重 (>1 = 强化学习, <1 = 减弱学习)

        Returns:
            被创建或更新的 Cluster
        """
        # 临时提升学习率以反映权重
        orig_lr = self.net.theta.learn_rate_l0
        self.net.theta.learn_rate_l0 = orig_lr * weight
        c = self.net.learn(fact_vec)
        self.net.theta.learn_rate_l0 = orig_lr
        self.n_facts += 1
        return c

    # ----- 查询 -----
    def query(self, query_vec: np.ndarray,
              top_k: int = 5) -> List[Tuple[Cluster, float]]:
        """查询语义知识。

        Args:
            query_vec: (D,) 查询向量 (通常只有 text[0:64] 段有效)
            top_k: 返回前 k 个匹配

        Returns:
            [(cluster, similarity), ...] 按相似度降序
        """
        if self.net.n_clusters == 0:
            return []

        if len(query_vec) < D:
            padded = np.zeros(D, dtype=np.float32)
            padded[:len(query_vec)] = query_vec
            query_vec = padded

        mask = _auto_mask(query_vec)
        h = self.net.hash_features(query_vec)

        # 扫描所有簇 (语义记忆中跨桶查询 — 概念可能在不同桶)
        scored = []
        for c in self.net.clusters:
            sim = _masked_cosine(h, c.centroid, mask)
            if sim > 0.15:  # 极低阈值 — 任何微弱关联都返回
                scored.append((c, float(sim)))

        scored.sort(key=lambda x: x[1], reverse=True)
        return scored[:top_k]

    def knows_about(self, query_vec: np.ndarray) -> float:
        """返回 Agent 对某主题的"熟悉度" [0, 1]。

        0 = 完全不知道, 1 = 非常熟悉 (多个强关联簇)。
        """
        results = self.query(query_vec, top_k=3)
        if not results:
            return 0.0
        # 熟悉度 = top-3 相似度的加权和
        familiarity = sum(sim * (0.7 ** i) for i, (_, sim)
                         in enumerate(results))
        return float(np.clip(familiarity / 1.5, 0.0, 1.0))

    # ----- 系统巩固: 情景→语义提取 -----
    def consolidate_from_episodic(self,
                                   episodic_net: ClusterNetwork,
                                   n_top: int = 30,
                                   min_activation: float = 0.1,
                                   body_vec=None,
                                   valence: float = 0.0,
                                   arousal: float = 0.0
                                   ) -> dict:
        """从海马情景记忆中提取主旨 → 存入语义记忆。

        这是系统巩固的核心操作: 将具体经验转化为抽象知识。
        只处理激活度最高的 N 个情景簇以控制成本。

        Args:
            episodic_net: 海马 ClusterNetwork
            n_top: 最多处理 N 个情景簇
            min_activation: 最低激活阈值
            body_vec: 当前身体向量 (用于情感门控)
            valence: 当前效价
            arousal: 当前唤醒

        Returns:
            dict: {n_processed, n_new, n_updated, n_skipped}
        """
        if episodic_net.n_clusters == 0:
            return {'n_processed': 0, 'n_new': 0, 'n_updated': 0, 'n_skipped': 0}

        # 选择激活度最高的簇
        candidates = sorted(
            [c for c in episodic_net.clusters
             if c.activation > min_activation],
            key=lambda c: c.activation, reverse=True,
        )[:n_top]

        stats = {'n_processed': 0, 'n_new': 0, 'n_updated': 0, 'n_skipped': 0}

        for c in candidates:
            stats['n_processed'] += 1
            # 情感门控: 高唤醒情景 → 更可能被语义化
            # 杏仁核情绪标记: 强烈情绪的记忆更容易转化为长期知识
            emotion_salience = 0.3 + 0.7 * abs(valence) * arousal
            if emotion_salience < 0.2 and stats['n_processed'] > 10:
                stats['n_skipped'] += 1
                continue

            # 提取主旨
            gist = self.extract_gist(c.centroid)

            # 注入当前情感/身体上下文
            if body_vec is not None and len(body_vec.b) >= 8:
                gist[64:72] = body_vec.b[:8].astype(np.float32)
            gist[72] = valence
            gist[73] = arousal

            # 检查是否已有类似知识
            existing = self.net.recall(gist)
            if existing is not None:
                # 更新已有知识 (慢速, 因为语义记忆需要多次确认)
                orig_lr = self.net.theta.learn_rate_l0
                # 已有知识的更新速率更低 (知识越牢固越难改变)
                stability = min(1.0, existing.count / 20.0)
                self.net.theta.learn_rate_l0 = orig_lr * (1.0 - stability * 0.8)
                self.net.learn(gist)
                self.net.theta.learn_rate_l0 = orig_lr
                stats['n_updated'] += 1
            else:
                # 新知识存储
                self.learn_fact(gist, weight=1.5)  # 首次提取给较高权重
                stats['n_new'] += 1

        self.consolidation_count += 1
        return stats

    # ----- 统计 -----
    @property
    def n_clusters(self) -> int:
        return self.net.n_clusters

    @property
    def total_activation(self) -> float:
        return self.net.total_activation

    def get_state(self) -> dict:
        """返回语义记忆状态摘要."""
        top = self.net.get_top_clusters(5)
        return {
            'n_clusters': self.n_clusters,
            'n_facts': self.n_facts,
            'consolidation_count': self.consolidation_count,
            'total_activation': self.total_activation,
            'top_clusters': [
                {'activation': float(c.activation),
                 'count': c.count, 'age': c.age}
                for c in top
            ],
        }
```

### Task 3.1: 重写 striatum.py

**Files:**

- [ ] **Step 1: 重写 striatum.py**

完整重写文件内容:

```python
"""
striatum.py — 纹状体 (Striatum)  [v6.0 完整实现]

对应脑区: 尾状核 (Caudate) + 壳核 (Putamen) + 伏隔核 (NAc)
所属层级: 大脑 → 基底神经节 → 纹状体

DMS (背内侧纹状体 = 尾状核): 目标导向行为, A→O 学习
DLS (背外侧纹状体 = 壳核后部): 习惯行为, S→R 学习
NAc (伏隔核): incentive salience, 多巴胺门控

功能职责:
  - D1 直接通路 (Go):  感知状态 → 促进特定动作 (多巴胺↑ → 增强)
  - D2 间接通路 (No-Go): 感知状态 → 抑制特定动作 (多巴胺↓ → 增强)
  - 习惯自动化: 重复的 S→R 映射从 DMS 转移至 DLS
  - 去价值化诊断: 检测行为是否已习惯化

存储机制:
  与海马情景记忆不同，纹状体不使用 ClusterNetwork。
  程序性记忆使用 S-R 联结表 — 稀疏的 dict 映射。
  这对应基底节的解剖: 中型多棘神经元 (MSN) 的突触权重矩阵。

参考:
  - Yin & Knowlton (2006). The role of the basal ganglia in habit formation.
  - Graybiel, A. M. (2008). Habits, rituals, and the evaluative brain.
"""

import numpy as np
from typing import Optional, Tuple, Dict


def _state_hash(state_vec: np.ndarray, n_bits: int = 12) -> int:
    """将感知状态哈希为整数键。

    使用前 n_bits 维度的符号位 → n_bits-bit 整数。
    这提供了粗糙但有辨别力的状态离散化。
    """
    if len(state_vec) < n_bits:
        padded = np.zeros(n_bits, dtype=np.float32)
        padded[:len(state_vec)] = state_vec[:len(state_vec)]
        state_vec = padded
    bits = (state_vec[:n_bits] > 0).astype(int)
    return int(sum(bits[i] << i for i in range(n_bits)))


class Striatum:
    """纹状体 — 程序性记忆与习惯学习。

    两个并行通路:
      D1 (直接/Go):   状态 → 促进动作, 多巴胺↑ → 增强
      D2 (间接/No-Go): 状态 → 抑制动作, 多巴胺↓ → 增强

    习惯从 DMS (灵活, 目标导向) 向 DLS (僵化, 自动化) 渐进转移。
    """

    def __init__(self, n_actions: int = 5, d1_d2_balance: float = 0.5):
        self.n_actions = n_actions
        self.d1_d2_balance = d1_d2_balance  # 0=D2主导, 1=D1主导

        # D1 直接通路: {state_hash → np.array(n_actions)} — 动作促进强度
        self.d1_weights: Dict[int, np.ndarray] = {}

        # D2 间接通路: {state_hash → np.array(n_actions)} — 动作抑制强度
        self.d2_weights: Dict[int, np.ndarray] = {}

        # 习惯强度追踪: {state_hash → float [0,1]} — 该状态的习惯化程度
        self.habit_strength: Dict[int, float] = {}

        # 状态-动作共现计数: {(state_hash, action) → int}
        self.cooccurrence: Dict[Tuple[int, int], int] = {}

        # 统计
        self.n_learn_events: int = 0
        self.n_habitual_actions: int = 0
        self.total_habit_strength: float = 0.0
        self._habit_ema: float = 0.0  # 全局习惯化 EMA

    # ================================================================
    # 学习
    # ================================================================

    def learn(self, state_vec: np.ndarray, action: int,
              da_signal: float, reward: float = 0.0,
              automation_rate: float = 0.05):
        """一次试错学习事件。

        Args:
            state_vec: 感知状态 (D,) 或 (N,)
            action: 执行的动作索引 [0, n_actions-1]
            da_signal: 多巴胺信号 [-1, 1] — 正=奖励预测误差>0, 负=RPE<0
            reward: 环境奖励 [0, 1]
            automation_rate: 习惯自动化速度
        """
        sh = _state_hash(state_vec)
        self.n_learn_events += 1

        # 确保权重数组存在
        if sh not in self.d1_weights:
            self.d1_weights[sh] = np.zeros(self.n_actions, dtype=np.float32)
        if sh not in self.d2_weights:
            self.d2_weights[sh] = np.zeros(self.n_actions, dtype=np.float32)

        # ---- D1 直接通路 (Go): 多巴胺↑ → 增强动作 ----
        if da_signal > 0:
            # D1 MSN: 多巴胺通过D1受体增强直接通路
            lr_d1 = 0.1 * da_signal * self.d1_d2_balance
            self.d1_weights[sh][action] += lr_d1 * (1.0 - self.d1_weights[sh][action])
        else:
            # 多巴胺↓ → D1 通路轻微衰减
            self.d1_weights[sh][action] *= 0.98

        # ---- D2 间接通路 (No-Go): 多巴胺↓ → 抑制动作 ----
        if da_signal < 0:
            # D2 MSN: 多巴胺下降解除D2抑制 → 增强间接通路
            lr_d2 = 0.1 * abs(da_signal) * (1.0 - self.d1_d2_balance)
            self.d2_weights[sh][action] += lr_d2 * (1.0 - self.d2_weights[sh][action])
        else:
            # 多巴胺↑ → D2 通路衰减 (多巴胺抑制D2 MSN)
            self.d2_weights[sh][action] *= 0.98

        # ---- 共现计数 (用于习惯强度计算) ----
        key = (sh, action)
        self.cooccurrence[key] = self.cooccurrence.get(key, 0) + 1

        # ---- 更新习惯强度 ----
        # 习惯强度 = 该状态-动作对的重复次数比例
        total_actions_in_state = sum(
            self.cooccurrence.get((sh, a), 0)
            for a in range(self.n_actions))
        if total_actions_in_state > 0:
            cooc = self.cooccurrence.get(key, 0)
            # 习惯化 = 该动作在该状态下的比例 (如果某个动作总是被选 = 高度习惯化)
            action_ratio = cooc / max(total_actions_in_state, 1)
            # 结合重复次数的渐进函数
            repetition_factor = 1.0 - np.exp(-total_actions_in_state / 20.0)
            new_habit = action_ratio * repetition_factor

            if sh not in self.habit_strength:
                self.habit_strength[sh] = new_habit
            else:
                # EMA 更新习惯强度
                self.habit_strength[sh] += automation_rate * (
                    new_habit - self.habit_strength[sh])

        # 全局习惯化 EMA
        if self.habit_strength:
            self._habit_ema += 0.01 * (
                np.mean(list(self.habit_strength.values())) - self._habit_ema)

    # ================================================================
    # 动作选择
    # ================================================================

    def get_habit_action(self, state_vec: np.ndarray,
                         novelty: float = 0.0) -> Optional[int]:
        """从习惯 (DLS) 中选择动作。

        Args:
            state_vec: 感知状态
            novelty: 当前新颖性信号 [0,1] — 高新颖性抑制习惯

        Returns:
            习惯动作索引, 或 None (无习惯)
        """
        sh = _state_hash(state_vec)
        hs = self.habit_strength.get(sh, 0.0)

        # 新颖性抑制习惯 (高新颖性 → 需要灵活行为, 抑制习惯)
        effective_habit = hs * (1.0 - novelty * 0.7)

        # 习惯强度不足以自动化
        if effective_habit < 0.3:
            return None

        if sh not in self.d1_weights or sh not in self.d2_weights:
            return None

        # Go - NoGo 净值 = 动作倾向
        net_strength = (self.d1_weights[sh]
                       - self.d2_weights[sh] * (1.0 - self.d1_d2_balance))

        # 选择净强度最高的动作
        best_action = int(np.argmax(net_strength))
        if net_strength[best_action] < 0.1:
            return None

        self.n_habitual_actions += 1
        return best_action

    def get_goal_directed_action(self, state_vec: np.ndarray) -> np.ndarray:
        """获取 DMS 目标导向的动作倾向分布。

        Returns:
            (n_actions,) 动作概率分布
        """
        sh = _state_hash(state_vec)
        if sh not in self.d1_weights:
            return np.ones(self.n_actions) / self.n_actions

        go = self.d1_weights.get(sh, np.zeros(self.n_actions))
        nogo = self.d2_weights.get(sh, np.zeros(self.n_actions))
        net = go - nogo * 0.5

        # softmax 转换为概率
        net = np.clip(net, -5, 5)
        exp = np.exp(net - np.max(net))
        probs = exp / (exp.sum() + 1e-8)
        return probs.astype(np.float64)

    # ================================================================
    # 诊断
    # ================================================================

    def is_habitual(self, state_vec: np.ndarray) -> bool:
        """检测给定状态下的行为是否已习惯化。"""
        sh = _state_hash(state_vec)
        return self.habit_strength.get(sh, 0.0) > 0.7

    def devaluation_test(self, state_vec: np.ndarray, action: int,
                         da_signal_on_devaluation: float = -0.5) -> bool:
        """去价值化诊断: 模拟结果贬值后行为是否仍持续。

        这是习惯的操作性定义 — 对结果价值变化不敏感。

        Args:
            state_vec: 感知状态
            action: 被测试的动作
            da_signal_on_devaluation: 贬值后的多巴胺信号 (负=不好)

        Returns:
            True = 习惯行为 (结果贬值后仍选择同一动作)
            False = 目标导向 (结果贬值后行为减少)
        """
        sh = _state_hash(state_vec)
        hs = self.habit_strength.get(sh, 0.0)

        # 贬值前: D1 对该动作的强度
        d1_before = self.d1_weights.get(sh, np.zeros(self.n_actions))[action]

        # 模拟贬值: 用负面多巴胺信号学习一次
        self.learn(state_vec, action, da_signal_on_devaluation, reward=0.0)

        # 贬值后: 习惯行为 (高习惯强度) → d1 几乎不变
        #         目标导向 (低习惯强度) → d1 显著下降
        d1_after = self.d1_weights[sh][action]
        d1_change = abs(d1_after - d1_before)

        # 恢复 (去价值化测试不应永久改变权重)
        self.d1_weights[sh][action] = d1_before

        # 习惯: d1 变化小 (< 0.05)
        return d1_change < 0.05

    @property
    def global_habit_strength(self) -> float:
        """全局习惯化程度 [0,1] — 整个纹状体的自动化水平。"""
        return float(self._habit_ema)

    def get_state(self) -> dict:
        """返回纹状体状态摘要。"""
        return {
            'n_states_known': len(self.d1_weights),
            'n_learn_events': self.n_learn_events,
            'n_habitual_actions': self.n_habitual_actions,
            'global_habit_strength': self.global_habit_strength,
            'n_habitual_states': sum(
                1 for hs in self.habit_strength.values() if hs > 0.5),
            'd1_d2_balance': self.d1_d2_balance,
        }
```

**Files:**
- Create: `D:\NotMe\cerebrum\parietal_lobe\working_memory.py`

**Purpose:** 实现 Baddeley 模型中的视空间模板——视觉图像和空间信息的暂存与操作。

- [ ] **Step 1: 创建 working_memory.py**

```python
"""
working_memory.py — 视空间模板 (Visuospatial Sketchpad)

脑区映射:
  - 右侧顶下皮层: 视觉暂存
  - 顶内沟 (IPS): 空间位置操作
  - 右侧额下外侧PFC: 主动复述

功能:
  - 暂存视觉图像信息 (最多4个组块, ~2秒消退)
  - 空间信息操作 (心理旋转, 位置比较)
  - 主动复述维持 (需注意力资源)

Baddeley 工作记忆模型的视空间子系统。
右半球优势 — 与左半球语音回路功能对称。
"""

import numpy as np
from typing import Optional, List, Tuple


class VisuospatialSketchpad:
    """视空间模板 — 视觉和空间信息的暂存与操作。

    容量: ~4 个视觉组块 (比语音回路的 7±2 更少)
    消退: ~2 秒无复述时信息衰减
    操作: 心理旋转, 空间关系比较
    """

    def __init__(self, max_chunks: int = 4, decay_rate: float = 0.3):
        self.max_chunks = max_chunks
        self.decay_rate = decay_rate

        # 视觉存储: [(visual_vector, spatial_tag, timestamp)]
        # visual_vector: (64,) 视觉特征
        # spatial_tag: (32,) 空间位置信息
        self.visual_store: List[Tuple[np.ndarray, np.ndarray, float]] = []

        # 空间工作区: (32,) 当前注意力焦点的空间位置
        self.spatial_focus: np.ndarray = np.zeros(32, dtype=np.float32)

        # 统计
        self.n_operations: int = 0  # 已执行的空间操作数

    # ----- 存储 -----

    def sketch(self, visual_input: np.ndarray,
               spatial_info: np.ndarray = None,
               timestamp: float = 0.0):
        """暂存一个视觉组块。

        Args:
            visual_input: (64,) 视觉特征向量
            spatial_info: (32,) 空间位置信息, None则用默认
            timestamp: 时间戳 (用于消退计算)
        """
        if len(visual_input) > 64:
            visual_input = visual_input[:64]

        if spatial_info is None:
            spatial_info = np.zeros(32, dtype=np.float32)
        elif len(spatial_info) > 32:
            spatial_info = spatial_info[:32]

        vis_vec = np.asarray(visual_input, dtype=np.float32).copy()
        sp_vec = np.asarray(spatial_info, dtype=np.float32).copy()

        # 容量管理: 溢出时替换最旧的组块
        if len(self.visual_store) >= self.max_chunks:
            self.visual_store.pop(0)

        self.visual_store.append((vis_vec, sp_vec, timestamp))

    def refresh(self, timestamp: float):
        """衰减过期组块并保持<=max_chunks。

        模拟组块的被动消退 — 无主动复述时信息丢失。
        """
        if not self.visual_store:
            return

        # 移除过期的组块 (超过 ~2秒无复述)
        kept = []
        for vis, sp, ts in self.visual_store:
            age = timestamp - ts
            # 指数衰减: 超过 2 秒后激活降至 0.37 以下
            if age < 5.0 and np.random.random() > (1.0 - np.exp(-self.decay_rate * age)):
                kept.append((vis, sp, ts))
        self.visual_store = kept

    # ----- 操作 -----

    def mental_rotation(self, degrees: float,
                       chunk_index: int = 0) -> Optional[np.ndarray]:
        """心理旋转: 对指定组块进行心理旋转。

        模拟空间工作记忆中的心理操作。
        旋转通过循环平移空间标签实现。

        Args:
            degrees: 旋转角度
            chunk_index: 要旋转的组块索引

        Returns:
            旋转后的空间向量, 或 None (组块不存在)
        """
        if chunk_index >= len(self.visual_store):
            return None

        vis, sp, ts = self.visual_store[chunk_index]
        # 旋转 = 空间标签的循环平移
        shift = int((degrees / 360.0) * len(sp))
        rotated = np.roll(sp.copy(), shift)
        self.n_operations += 1
        return rotated

    def spatial_compare(self, target: np.ndarray) -> float:
        """比较当前空间焦点与目标的距离。

        Args:
            target: (32,) 目标空间位置

        Returns:
            余弦相似度 [-1, 1]
        """
        if len(target) > 32:
            target = target[:32]
        denom = (np.linalg.norm(self.spatial_focus)
                 * np.linalg.norm(target) + 1e-8)
        return float(np.dot(self.spatial_focus, target) / denom)

    # ----- 查询 -----

    def get_active_visuals(self) -> List[np.ndarray]:
        """返回当前活跃的视觉组块 (按时间降序)."""
        return [vis for vis, sp, ts in
                sorted(self.visual_store, key=lambda x: x[2], reverse=True)]

    @property
    def n_chunks(self) -> int:
        return len(self.visual_store)

    @property
    def is_full(self) -> bool:
        return self.n_chunks >= self.max_chunks

    def get_state(self) -> dict:
        return {
            'n_chunks': self.n_chunks,
            'max_chunks': self.max_chunks,
            'n_operations': self.n_operations,
            'spatial_focus_norm': float(np.linalg.norm(self.spatial_focus)),
        }
```

```python
    def __init__(self, input_dim: int = 516):
        self.input_dim = input_dim
        self.attention_template: np.ndarray = np.ones(input_dim, dtype=np.float32)
        self._prev_template: np.ndarray = np.ones(input_dim, dtype=np.float32)

        # v6.0: 中央执行器状态
        self.task_goal: np.ndarray = np.zeros(64, dtype=np.float32)  # 当前任务目标
        self.working_memory_load: float = 0.0   # 工作记忆负载 [0, 1]
        self.distractor_suppression: float = 0.3  # 干扰抑制强度 [0, 1]
        self.task_switch_cost: float = 0.0      # 任务切换代价 [0, 1]

    def allocate_attention(self, task_demand: float,
                          wm_load: float = 0.0,
                          novelty: float = 0.0) -> dict:
        """中央执行器: 注意力资源分配。

        有限的注意力资源在多个子系统间分配。
        高任务需求 → 更多资源给当前任务
        高新颖性 → 资源重新定向到新刺激
        高 WM 负载 → 减少新信息摄入

        Args:
            task_demand: 任务需求强度 [0, 1]
            wm_load: 工作记忆负载 [0, 1]
            novelty: 新颖性信号 [0, 1]

        Returns:
            dict: {attn_focus, suppression_level, switch_triggered}
        """
        self.working_memory_load = wm_load

        # 基础注意力焦点 = 任务需求
        attn_focus = task_demand

        # WM 负载调制: 负载高 → 减少新信息摄入
        effective_attn = attn_focus * (1.0 - wm_load * 0.5)

        # 新颖性调制: 新高新颖 → 可能触发注意切换
        switch_triggered = False
        if novelty > 0.6 and task_demand < 0.5:
            # 新颖刺激出现 + 当前任务需求不高 → 切换注意
            effective_attn = novelty * 0.8
            self.task_switch_cost = 0.3  # 切换代价
            switch_triggered = True
        else:
            # 切换代价逐渐消退
            self.task_switch_cost *= 0.9

        # 干扰抑制强度 = f(任务专注度)
        self.distractor_suppression = 0.2 + 0.6 * task_demand

        return {
            'attn_focus': float(effective_attn),
            'suppression_level': float(self.distractor_suppression),
            'switch_triggered': switch_triggered,
            'switch_cost': float(self.task_switch_cost),
        }

    def filter_distractors(self, sensory: np.ndarray) -> np.ndarray:
        """抑制与当前任务目标无关的感觉输入。

        基于当前注意力模板和干扰抑制强度。
        非目标维度被衰减 (但不完全归零 — 保留对突发威胁的响应)。

        Args:
            sensory: (D,) 原始感觉输入

        Returns:
            (D,) 过滤后的感觉输入
        """
        # 构建抑制掩码: 注意力模板低的维度被抑制
        template = self.attention_template.copy()
        # 模板归一化到 [distractor_suppression, 1.0] 范围
        # 最低保留 distractor_suppression 比例 (不完全阻断)
        suppress_mask = (template - template.min()) / (
            template.max() - template.min() + 1e-8)
        suppress_mask = (self.distractor_suppression
                        + (1.0 - self.distractor_suppression) * suppress_mask)

        return sensory * suppress_mask.astype(np.float32)

    def coordinate_subsystems(self,
                             phonological_load: float = 0.0,
                             visuospatial_load: float = 0.0,
                             episodic_buffer_load: float = 0.0) -> dict:
        """协调工作记忆子系统——分配注意力资源。

        语音回路和视空间模板竞争注意力资源。
        中央执行器决定优先级。

        Returns:
            dict: 各子系统的注意力权重
        """
        total_load = phonological_load + visuospatial_load + episodic_buffer_load
        if total_load < 0.01:
            return {'phonological': 0.33, 'visuospatial': 0.33,
                   'episodic_buffer': 0.34}

        return {
            'phonological': phonological_load / max(total_load, 0.01),
```

### Task 5.2: 杏仁核增强 — 情感条件作用

**Files:**
- Modify: `D:\NotMe\cerebrum\limbic_system\amygdala.py`

**Purpose:** 在已有 Hebb 情感词汇学习基础上，加入简单的恐惧条件作用——将中性刺激与负面身体状态关联。

- [ ] **Step 1: 添加恐惧条件作用**

在 `HebbEmotionalLexicon` 类中添加:

```python
    def __init__(self, dim: int = 64, learning_rate: float = 0.05):
        # ... existing init ...
        # v6.0: 条件作用追踪
        self.conditioned_stimuli: dict[str, float] = {}  # {word_hash → fear_association}
        self._fear_threshold: float = 0.5  # 恐惧激活阈值

    def condition_fear(self, stimulus_vec: np.ndarray,
                      fear_intensity: float):
        """经典恐惧条件作用: 将之前中性的刺激与高唤醒/负面状态关联。

        一次高强度负面体验即可形成持久的恐惧记忆 (快速学习)。
        这模拟杏仁核-外侧杏仁核 (LA) 的可塑性 — 恐惧记忆的核心底物。

        Args:
            stimulus_vec: (64,) 刺激的语义向量
            fear_intensity: 恐惧强度 [0, 1] — 基于 F_body↑ + valence↓
        """
        if fear_intensity < 0.3:
            return  # 不够强烈, 不形成条件作用

        # 哈希刺激 (用向量前 16 维的符号位)
        stim_key = ''.join('1' if v > 0 else '0'
                          for v in stimulus_vec[:16])
        # 快速学习: 一次高强度暴露即可
        if stim_key in self.conditioned_stimuli:
            # 增强已有恐惧
            self.conditioned_stimuli[stim_key] = min(
                1.0,
                self.conditioned_stimuli[stim_key] + fear_intensity * 0.3)
        else:
            self.conditioned_stimuli[stim_key] = fear_intensity * 0.5

    def get_fear_response(self, stimulus_vec: np.ndarray) -> float:
        """获取对刺激的恐惧反应 [0, 1].

        0 = 无恐惧, 1 = 强恐惧。
        """
        stim_key = ''.join('1' if v > 0 else '0'
                          for v in stimulus_vec[:16])
        fear = self.conditioned_stimuli.get(stim_key, 0.0)
        return float(fear)

    def is_fear_conditioned(self, stimulus_vec: np.ndarray) -> bool:
        """检测刺激是否已形成恐惧条件作用."""
        return self.get_fear_response(stimulus_vec) > self._fear_threshold
```
